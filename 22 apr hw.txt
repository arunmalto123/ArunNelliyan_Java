import java.util.Scanner;
class SecondMax{
	public static void main (String [] args){
		Scanner sc = new Scanner(System.in);
		System.out.println("Enter the total strength:");
		int strength=sc.nextInt();
		int arr[]=new int[strength];
		System.out.println("Enter the values:  ");
		for(int i=0;i<arr.length;i++)
		{
			System.out.println("values of :"+(i+1)+"index  is :");
			arr[i]=sc.nextInt();
		}
		int max=0;
		int secondMax=0;
		for(int i=0;i<arr.length;i++)
		{
			if(arr[i]>secondMax)
			{
				secondMax=max;
				max=arr[i];
			}else if(arr[i]>secondMax && arr[i]!= max)
			{
				secondMax =arr[i];
			}
		}
		
	System.out.println ("second max value is :"+secondMax);
		

	}
}

import java.util.Scanner;
class Shopping
{
 public static void main(String [] args)
  {
   Scanner sc=new Scanner(System.in);
   System.out.println("enter total number of items:");
   int itemCount=sc.nextInt();

   int prices[]=new int[itemCount];
   int total=0;

    for(int i=0;i<itemCount;i++)
	{
		 System.out.print("Enter price of item " + (i + 1) + ": ");
		 prices[i]=sc.nextInt();
		 total = total+prices[i];
	}		
	
	double discount=0;
	if( total > 5000)
	{
		discount =0.20;
	}
	else if( total >2000)
	{
		discount = 0.10;
	}
	else 
	{
		System.out.println("no discount");
	}
	
	double  discountRate = total * discount;
        double  finalBill = total - discountRate;
		
   System.out.println("\n------ BILL DETAILS ------");
        System.out.println("Total Amount : " + total);
        System.out.println("Discount     : " + discountRate);
        System.out.println("Final Amount : " + finalBill);
   
  }
}
