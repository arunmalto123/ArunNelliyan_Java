import java.util.Scanner;
class oddEven
{
	public static void main (String [] args)
	{
		
		System.out.println("Enter number:");
		Scanner scanner =new Scanner(System.in);
		int num =scanner.nextInt();
		if (num % 2 == 0)
		{
			System.out.println("given num :"+ num + " is even number");
		}
		else 
		{
			System.out.println("given num :"+num+" is odd number");
		}
		
	}
}

import java.util.Scanner;
class posNeg
{
	public static void main (String [] args)
	{
		
		System.out.println("Enter number:");
		Scanner scanner =new Scanner(System.in);
		int num =scanner.nextInt();
		if (num > 0)
		{
			System.out.println("given num :"+ num + " is positive number");
		}
		else if(num < 0)
		{
			System.out.println("given num :"+num+" is negative number");
		}
		else 
		{
			System.out.println("given number is neither positive or negative");
		}
	}
}
